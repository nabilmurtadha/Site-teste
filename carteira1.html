  x <- xts(get(x)[,6]*carteira$qtd[carteira$acao == as.character(x)])
})

names(dados) <- carteira$acao
total <- dados %>% 
  as_tibble() %>% 
  unnest() %>% 
  xts(order.by = tempo)
  
total <- merge(total, rowSums(total))


rendimento_diario <- merge(dailyReturn(total$rowSums.total., type = "log"),
                    dailyReturn(BVSP$BVSP.Adjusted, type = "log"),
                    selic)

  as_tibble() %>%
